function calculateCaffeineClearance() { const key = getCaffeineLogKey(); const timestamps = JSON.parse(localStorage.getItem(key) || "[]"); document.getElementById('coffee_count').innerText = timestamps.length; const badge = document.getElementById('clearance_status'); if (timestamps.length === 0) { badge.innerText = "0 mg"; badge.className = "clearance-badge bg-green"; return; } const CAFFEINE_PER_CUP = 80; const HALF_LIFE_HOURS = 5.0; // Construct fixed target 18:00 timestamp for the selected day const targetDate18 = new Date(datePicker.value + 'T18:00:00'); const time18 = targetDate18.getTime(); let caffeineAt18 = 0; let currentSystemCaffeine = 0; const now = new Date().getTime(); // Calculate precise decay levels timestamps.forEach(ts => { // 1. Level at 18:00 if (time18 >= ts) { let hrs = (time18 - ts) / 3600000; caffeineAt18 += CAFFEINE_PER_CUP * Math.pow(0.5, hrs / HALF_LIFE_HOURS); } else { // If coffee was logged AFTER 18:00, the full amount is present at 18:00 baseline evaluations caffeineAt18 += CAFFEINE_PER_CUP; } // 2. Live System Level Right Now if (now >= ts) { let hrs = (now - ts) / 3600000; currentSystemCaffeine += CAFFEINE_PER_CUP * Math.pow(0.5, hrs / HALF_LIFE_HOURS); } }); // Display current metabolic load badge.innerText = `${Math.round(currentSystemCaffeine)} mg live`; // Evaluate thresholds strictly against the 18:00 load parameters if (caffeineAt18 <= 20) { // 20mg or less -> Green badge.className = "clearance-badge bg-green"; badge.title = `Est at 18:00: ${Math.round(caffeineAt18)}mg`; } else if (caffeineAt18 <= 40) { // 21mg to 40mg -> Yellow/Orange badge.className = "clearance-badge bg-orange"; badge.title = `Est at 18:00: ${Math.round(caffeineAt18)}mg`; } else { // Over 40mg -> Red badge.className = "clearance-badge bg-red"; badge.title = `Est at 18:00: ${Math.round(caffeineAt18)}mg`; } }
